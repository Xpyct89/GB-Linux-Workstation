Задачи:

1. Вывести на экран 3 раза имя пользователя, от которого запускается команда.
2. Вывести с помощью цикла while все четные числа от 0 до 100 включительно.
3. Создать с помощью nano файл test.txt. Настроить автоматический бэкап этого файла раз в 10 минут в файл с названием test.txt.bak с использованием cron.

# Формируем цикл в самой строке терминала
for user in {1..3}; do echo $USER; done

# Создаем файл скрипта
nano task2.sh

# В файл скрипта записываем инструкцию
!/bin/bash
x=0
while [ $x -lt 102 ]
do
printf "${x} "
let x+=2
done
echo ""

# Разрешаем выполнение для владельца
chmod u+x task2.sh

# Проверяем доступы файла
ll

# Запускаем скрипт
./task2.sh

# Создаю файл
echo Some text > test.txt

# Запускаю редактирование планировщика cron
crontab -e

# Заносим скрипт в планировщик
*/10 * * * * cat /home/xpyct/test.txt > /home/xpyct/test.txt.bak

# Ждем 10 минут и проверяем исполнение скрипта
ls
